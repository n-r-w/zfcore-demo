4//
5// A pair of smart pointer template classes. Provides basic conversion
6// operator to T*, as well as dereferencing (*), and 0-checking (!).
7// These classes assume that they alone are responsible for deleting the
8// object or array unless Relinquish() is called.
9//
10
11//
12// class DPtrBase
13// ~~~~~ ~~~~~~~~~~~~
14template<class T> class DPtrBase
15{
16public:
17 T& operator *() {return *P;}
18 const T& operator *() const {return *P;}
19
20 operator T*() {return P;}
21 operator const T*() const {return P;}
22
23 T* GetPtr(void){return P;}
24 const T* GetPtr(void) const {return P;}
25
26 T** GetPtrPtr(void){return &P;}
27 T*& GetPtrRef(void){return P;}
28
29 int operator !() const {return P==0;}
30 T* Relinquish(T* newp=0) {T* p = P; P = newp; return p;}
31
32protected:
33 DPtrBase(T* pointer) : P(pointer) {}
34 DPtrBase() : P(0) {}
35
36 T* P = nullptr;
37
38#ifndef MPATROL_VERSION // *nix bounds checker
39private:
40 void* operator new(size_t) {return 0;} // prohibit use of new
41 #ifdef __GNUC__
42 // in gcc operator delete seems to be needed for destructors in derived classes,
43 // so it can't be private
44protected:
45 #endif // __GNUC__
46 void operator delete(void* p) {((DPtrBase<T>*)p)->P=0;}
47 #endif // MPATROL_VERSION
48private:
49 DPtrBase(const DPtrBase<T> &);
50 void operator = (const DPtrBase<T> &);
51};
