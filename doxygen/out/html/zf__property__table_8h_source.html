78 QList<Entity> res(_entities.keys());
79 std::sort(res.begin(), res.end(), [&](const Entity& e1, const Entity& e2) -> bool {
80 int order1 = _entities.value(e1);
81 int order2 = _entities.value(e2);
82 if (order1 >= 0 && order2 >= 0)
83 return (order == Qt::AscendingOrder) ? order1 < order2 : order1 > order2;
84
85 if (order1 < 0 && order2 < 0)
86 return (order == Qt::AscendingOrder) ? e1 < e2 : e1 > e2;
87
88 if (order1 >= 0)
89 return (order == Qt::AscendingOrder) ? true : false;
90

147 for (auto it = values.constBegin(); it != values.constEnd(); ++it) {

177 void setValue(const Entity& id, const Property& property, const QVariant& value, int order = -1,
178 QLocale::Language language = QLocale::AnyLanguage)
179 {
180 QString key = dataKey(id, property, _ignore_set_error);
181 if (key.isEmpty()) {
182 if (!_auto_add_properties && _ignore_set_error)
183 return;
184 if (_auto_add_properties) {
185 addProperties({property});
186 key = dataKey(id, property, false);
187 }
188 }
189
190 auto it = _data.find(key);
191 if (it == _data.end()) {
192 _data[key] = LanguageMap({{language, value.isNull() ? QVariant() : value}});
193 } else {
194 it.value().insert(language, value.isNull() ? QVariant() : value);
195 }
196
197 _entities.insert(id, order);
198 clearCache();
199 }

202 void setValue(const Entity& id, const Property& property, const LanguageMap& value, int order = -1)

207 return;
208 if (_auto_add_properties) {

210 key = dataKey(id, property, false);
211 }
212 }
213

215 if (it == _data.end()) {
216 _data[key] = value;
217 } else {
218 it.value() = value;
219 }

257 void createCache() const
258 {
259 if (_data.isEmpty() || !_find_hash.isEmpty())
260 return;
261
262 for (auto ent = _entities.constBegin(); ent != _entities.constEnd(); ++ent) {
263 QMap<Property, QVariant> values_by_id = this->values(ent.key());
264 for (auto it = values_by_id.constBegin(); it != values_by_id.constEnd(); ++it) {
265 _find_hash.insert(hashKey(it.key(), it.value()), ent.key());
266 }
267 }
