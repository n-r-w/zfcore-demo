38#define DEBUG_BREAK_USE_TRAP_INSTRUCTION 1
39#define DEBUG_BREAK_USE_BULTIN_TRAP 2
40#define DEBUG_BREAK_USE_SIGTRAP 3
41
42#if defined(__i386__) || defined(__x86_64__)
43 #define DEBUG_BREAK_IMPL DEBUG_BREAK_USE_TRAP_INSTRUCTION
44__inline__ static void trap_instruction(void)
45{
46 __asm__ volatile("int $0x03");
47}
48#elif defined(__thumb__)
49 #define DEBUG_BREAK_IMPL DEBUG_BREAK_USE_TRAP_INSTRUCTION
50/* FIXME: handle __THUMB_INTERWORK__ */
51__attribute__((gnu_inline, always_inline))
52__inline__ static void trap_instruction(void)
53{
54 /* See 'arm-linux-tdep.c' in GDB source.
55 * Both instruction sequences below work. */
56#if 1
57 /* 'eabi_linux_thumb_le_breakpoint' */
58 __asm__ volatile(".inst 0xde01");
59#else
60 /* 'eabi_linux_thumb2_le_breakpoint' */
61 __asm__ volatile(".inst.w 0xf7f0a000");
62#endif
63
64 /* Known problem:
65 * After a breakpoint hit, can't 'stepi', 'step', or 'continue' in GDB.
66 * 'step' would keep getting stuck on the same instruction.
67 *
68 * Workaround: use the new GDB commands 'debugbreak-step' and
69 * 'debugbreak-continue' that become available
70 * after you source the script from GDB:
71 *
72 * $ gdb -x debugbreak-gdb.py <... USUAL ARGUMENTS ...>
73 *
74 * 'debugbreak-step' would jump over the breakpoint instruction with
75 * roughly equivalent of:
76 * (gdb) set $instruction_len = 2
77 * (gdb) tbreak *($pc + $instruction_len)
78 * (gdb) jump *($pc + $instruction_len)
79 */
80}
81#elif defined(__arm__) && !defined(__thumb__)
82 #define DEBUG_BREAK_IMPL DEBUG_BREAK_USE_TRAP_INSTRUCTION
83__attribute__((gnu_inline, always_inline))
84__inline__ static void trap_instruction(void)
85{
86 /* See 'arm-linux-tdep.c' in GDB source,
87 * 'eabi_linux_arm_le_breakpoint' */
88 __asm__ volatile(".inst 0xe7f001f0");
89 /* Known problem:
90 * Same problem and workaround as Thumb mode */
91}
92#elif defined(__aarch64__) && defined(__APPLE__)
93 #define DEBUG_BREAK_IMPL DEBUG_BREAK_USE_BULTIN_TRAP
94#elif defined(__aarch64__)
95 #define DEBUG_BREAK_IMPL DEBUG_BREAK_USE_TRAP_INSTRUCTION
96__attribute__((gnu_inline, always_inline))
97__inline__ static void trap_instruction(void)
98{
99 /* See 'aarch64-tdep.c' in GDB source,
100 * 'aarch64_default_breakpoint' */
101 __asm__ volatile(".inst 0xd4200000");
102}
103#elif defined(__powerpc__)
104/* PPC 32 or 64-bit, big or little endian */
105#define DEBUG_BREAK_IMPL DEBUG_BREAK_USE_TRAP_INSTRUCTION
106__attribute__((gnu_inline, always_inline))
107__inline__ static void trap_instruction(void)
108{
109 /* See 'rs6000-tdep.c' in GDB source,
110 * 'rs6000_breakpoint' */
111 __asm__ volatile(".4byte 0x7d821008");
112
113 /* Known problem:
114 * After a breakpoint hit, can't 'stepi', 'step', or 'continue' in GDB.
115 * 'step' stuck on the same instruction ("twge r2,r2").
116 *
117 * The workaround is the same as ARM Thumb mode: use debugbreak-gdb.py
118 * or manually jump over the instruction. */
119}
120#else
121 #define DEBUG_BREAK_IMPL DEBUG_BREAK_USE_SIGTRAP
122#endif
123
124
125#ifndef DEBUG_BREAK_IMPL
126#error "debugbreak.h is not supported on this target"
127#elif DEBUG_BREAK_IMPL == DEBUG_BREAK_USE_TRAP_INSTRUCTION
128__attribute__((gnu_inline, always_inline)) __inline__ static void debug_break(void)
129{
130 trap_instruction();
131}
132#elif DEBUG_BREAK_IMPL == DEBUG_BREAK_USE_BULTIN_TRAP
133__attribute__((gnu_inline, always_inline))
134__inline__ static void debug_break(void)
135{
136 __builtin_trap();
137}
138#elif DEBUG_BREAK_IMPL == DEBUG_BREAK_USE_SIGTRAP
139#include <signal.h>
140__attribute__((gnu_inline, always_inline))
141__inline__ static void debug_break(void)
142{
143 raise(SIGTRAP);
144}
145#else
146#error "invalid DEBUG_BREAK_IMPL value"
147#endif
148
149#ifdef __cplusplus
150}
151#endif
152
153#endif /* ifdef _MSC_VER */
