193 template <class T>
194 inline T* field(const DataProperty& property,
196 bool halt_if_not_found = true,
198 bool halt_if_not_cast = true) const
199 {
200 QWidget* w = field(property, halt_if_not_found);
201 if (!halt_if_not_found && w == nullptr)
202 return nullptr;
203
204#ifdef Q_OS_LINUX
205#pragma GCC diagnostic push
206#pragma GCC diagnostic ignored "-Wnonnull"
207#endif
208
209 T* res = dynamic_cast<T*>(w);
210 if (halt_if_not_cast && res == nullptr) {
211 if (property.isEntityIndependent())
212 Z_HALT(QString("Type conversion error %1,%2: %3 -> %4")
213 .arg(data()->id())
214 .arg(property.id().value())
215 .arg(w->metaObject()->className())
216 .arg(T::staticMetaObject.className()));
217 else
218 Z_HALT(QString("Type conversion error %1,%2: %3 -> %4") //-V523
219 .arg(property.entityCode().value())
220 .arg(property.id().value())
221 .arg(w->metaObject()->className())
222 .arg(T::staticMetaObject.className()));
223 }
224
225#ifdef Q_OS_LINUX
226#pragma GCC diagnostic pop
227#endif
228
229 return res;
230 }
