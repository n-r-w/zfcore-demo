687 template <typename T>
688 static inline void shuffleList(QList<T>& a,
690 int seed = -1)
691 {
692 if (seed >= 0)
693 QRandomGenerator::system()->seed(static_cast<quint32>(seed));
694
695 int n = a.size();
696 for (int i = n - 1; i > 0; --i) { // gist, note, i>0 not i>=0
697 int r = QRandomGenerator::system()->generate() % (i + 1); // gist, note, i+1 not i. "rand() % (i+1)" means
698 // generate rand numbers from 0 to i
699 a.swap(i, r);
700 }
701 }
