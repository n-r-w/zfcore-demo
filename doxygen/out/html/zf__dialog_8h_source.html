192 template <class T>
193 T* object(
196 const QStringList& path) const
197 {
198 QList<QObject*> objects = findObjectsHelper(path, T::staticMetaObject.className());
199 Z_CHECK_X(!objects.isEmpty(),
200 QString("Object %1:%2 not found").arg(T::staticMetaObject.className()).arg(path.join("\\")));
201 Z_CHECK_X(objects.count() == 1, QString("Object %1:%2 found more then once (%4)")
202 .arg(T::staticMetaObject.className())
203 .arg(path.join("\\"))
204 .arg(objects.count()));
205 T* res = qobject_cast<T*>(objects.first());
206 Z_CHECK_X(res != nullptr,
207 QString("Casting error %1:%2").arg(T::staticMetaObject.className()).arg(path.join("\\")));
208
209 return res;
210 }
